void melhora(double t, char r[100]){
	sprintf(r,"%f",t);
	int x, tamanho=0,i,j;
  	tamanho=strlen(r);
  	for (x=0; x<tamanho; x++){
		if (r[x]== ',' || r[x]=='.'){
			r[x]='.';
			for(i=x;i<tamanho;i++){
				if(r[i]=='.'){
					j=x;
				}else if(r[i]!='0'){
					j=i+1;
				}
	 		}
	  	}
  	}
  	r[j]='\0';
}
